Binary Search 二分查找法

复杂度

时间

Average: \( O(log2(n)) \)

空间

\( O(1) \)

核心实现

给定已排序好的数组，在数组中查找对应值的下标，每次循环找到子数组中点判断大小。运行时间稳定。

function binarySearch(arr, target) {
  let low = 0,
    high = arr.length - 1; // low...high是当前正在搜索的子数组
  while (low <= high) {
    const mid = parseInt((low + high) * 0.5); // 计算中点
    if (target == arr[mid]) {
      return mid;
    } else if (target > arr[mid]) {
      low = mid + 1; // 提升low到mid
    } else if (target < arr[mid]) {
      high = mid - 1; // 降低high到mid
    }
  }
  return -1;
}

console.log(binarySearch([1, 3, 4, 5, 7, 9, 10, 11, 12, 14, 15, 19, 20], 4));
